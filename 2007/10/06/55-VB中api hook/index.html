Function HexIp2DotIp(ByVal ip As Long) As String
Dim s As String, p1 As String, p2 As String, p3 As String, p4 As String
s = Right(“00000000” & Hex(ip), 8)
p1 = Val(“&h” & Mid(s, 1, 2))
p2 = Val(“&h” & Mid(s, 3, 2))
p3 = Val(“&h” & Mid(s, 5, 2))
p4 = Val(“&h” & Mid(s, 7, 2))
HexIp2DotIp = p4 & “.” & p3 & “.” & p2 & “.” & p1
End Function
